Step 4: Setup Virtual Hosts
File: C:\xampp\apache\conf\extra\httpd-vhosts.conf
Add these virtual hosts at the end of the file:
apache# Marketing Website
<VirtualHost *:80>
    ServerName erp.local
    ServerAlias www.erp.local
    DocumentRoot "C:/xampp/htdocs/school-erp/public"
    
    <Directory "C:/xampp/htdocs/school-erp/public">
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted
    </Directory>
    
    ErrorLog "logs/erp-error.log"
    CustomLog "logs/erp-access.log" combined
</VirtualHost>

# Admin Panel
<VirtualHost *:80>
    ServerName admin.erp.local
    DocumentRoot "C:/xampp/htdocs/school-erp/public"
    
    <Directory "C:/xampp/htdocs/school-erp/public">
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted
    </Directory>
    
    ErrorLog "logs/admin-erp-error.log"
    CustomLog "logs/admin-erp-access.log" combined
</VirtualHost>

# Tenant Wildcard (All school subdomains)
<VirtualHost *:80>
    ServerName erp.local
    ServerAlias *.erp.local
    DocumentRoot "C:/xampp/htdocs/school-erp/public"
    
    <Directory "C:/xampp/htdocs/school-erp/public">
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted
    </Directory>
    
    ErrorLog "logs/tenant-erp-error.log"
    CustomLog "logs/tenant-erp-access.log" combined
</VirtualHost>
